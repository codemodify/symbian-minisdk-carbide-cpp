Cleanup Stack

void SafeFunctionL()
{
    CClanger* clanger = new (ELeave) CClanger;

    // push the it the cleanup stack in case the next call leaves
    CleanupStack::PushL( clanger );

    clanger->InitializeL();
    clanger->DoSomethingElseL()

    // pop from cleanup stack
    CleanupStack::Pop( clanger );

    delete clanger;
}
